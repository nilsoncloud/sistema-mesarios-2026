                document.body.classList.add('overflow-hidden');
            });

        function renderTasks() {
            todoList.innerHTML = '';
            
            if (tasks.length === 0) {
                todoList.innerHTML = `
                    <div class="text-center py-8 text-gray-500">
                        <i class="fas fa-clipboard-list text-4xl mb-3 opacity-50"></i>
                        <p>Nenhuma tarefa pendente</p>
                    </div>
                `;
                return;
            }
            
            tasks.forEach(task => {
                const taskItem = document.createElement('li');
                taskItem.className = `task-item bg-gray-50 rounded-lg p-4 flex items-center justify-between fade-in ${task.completed ? 'completed' : ''}`;
                taskItem.dataset.id = task.id;
                
                taskItem.innerHTML = `
                    <div class="flex items-center">
                        <button class="complete-btn mr-3 w-6 h-6 rounded-full border-2 ${task.completed ? 'bg-green-500 border-green-500 text-white' : 'border-gray-300'} flex items-center justify-center transition-colors">
                            ${task.completed ? '<i class="fas fa-check text-xs"></i>' : ''}
                        </button>
                        <span class="task-text ${task.completed ? 'line-through text-gray-500' : 'text-gray-800'}">${task.text}</span>
                    </div>
                    <button class="delete-btn text-red-500 hover:text-red-700 transition-colors">
                        <i class="fas fa-trash"></i>
                    </button>
                `;
                
                todoList.appendChild(taskItem);
                
                // Adicionar event listeners para os botões desta tarefa
                const completeBtn = taskItem.querySelector('.complete-btn');
                const deleteBtn = taskItem.querySelector('.delete-btn');
                
                completeBtn.addEventListener('click', function() {
                    toggleTaskCompletion(task.id);
                });
                
                deleteBtn.addEventListener('click', function() {
                    openDeleteModal(task.id);
                });
            });
        }
        
        function addNewTask() {
            const taskText = newTaskInput.value.trim();
            
            if (taskText === '') {
                newTaskInput.focus();
                return;
            }
            
            const newTask = {
                id: taskIdCounter++,
                text: taskText,
                completed: false
            };
            
            tasks.unshift(newTask);
            saveTasksToLocalStorage();
            renderTasks();
            updateTaskCounters();
            
            // Limpar o campo de entrada
            newTaskInput.value = '';
            newTaskInput.focus();
        }
        
        function toggleTaskCompletion(taskId) {
            const taskIndex = tasks.findIndex(task => task.id === taskId);
            
            if (taskIndex !== -1) {
                tasks[taskIndex].completed = !tasks[taskIndex].completed;
                saveTasksToLocalStorage();
                renderTasks();
                updateTaskCounters();
            }
        }
        
        function openDeleteModal(taskId) {
            taskToDelete = taskId;
            deleteModal.classList.add('active');
        }
        
        function closeDeleteModal() {
            deleteModal.classList.remove('active');
            taskToDelete = null;
        }
        
        function confirmDeleteTask() {
            if (taskToDelete) {
                tasks = tasks.filter(task => task.id !== taskToDelete);
                saveTasksToLocalStorage();
                renderTasks();
                updateTaskCounters();
                closeDeleteModal();
            }
        }
        
        function updateTaskCounters() {
            const totalTasks = tasks.length;
            const completedTasks = tasks.filter(task => task.completed).length;
            
            totalTasksSpan.textContent = `Total: ${totalTasks} tarefa${totalTasks !== 1 ? 's' : ''}`;
            completedTasksSpan.textContent = `Concluídas: ${completedTasks}`;
        }
        
        function saveTasksToLocalStorage() {
            localStorage.setItem('dashboardTasks', JSON.stringify(tasks));
        }
        
        function toggleTheme() {
            document.body.classList.toggle('dark-mode');
            
            if (document.body.classList.contains('dark-mode')) {
                localStorage.setItem('dashboardTheme', 'dark');
                themeToggle.innerHTML = '<i class="fas fa-sun text-yellow-500"></i>';
                document.body.style.backgroundColor = '#1a202c';
            } else {
                localStorage.setItem('dashboardTheme', 'light');
                themeToggle.innerHTML = '<i class="fas fa-moon text-gray-600"></i>';
                document.body.style.backgroundColor = '#f0f4f8';
            }
        }
        
        // ========== FUNÇÕES DO SISTEMA DE MESÁRIOS ==========
        function setupMesariosSystemEventListeners() {
            // Toggle da sidebar em mobile
            sidebarToggle.addEventListener('click', function() {
                document.querySelector('.sidebar').classList.toggle('mobile-open');
                mobileOverlay.classList.toggle('active');
            });
            
            // Fechar sidebar ao clicar no overlay
            mobileOverlay.addEventListener('click', function() {
                document.querySelector('.sidebar').classList.remove('mobile-open');
                mobileOverlay.classList.remove('active');
            });
            
            // Navegação do menu
            menuItems.forEach(item => {
                item.addEventListener('click', function(e) {
                    e.preventDefault();
                    const targetSection = this.getAttribute('data-section');
                    
                    // Esconder todas as seções
                    contentSections.forEach(section => {
                        section.classList.remove('active');
                        section.classList.add('hidden');
                    });
                    
                    // Mostrar a seção selecionada
                    document.getElementById(targetSection).classList.remove('hidden');
                    document.getElementById(targetSection).classList.add('active');
                    
                    // Fechar sidebar em mobile
                    if (window.innerWidth < 768) {
                        document.querySelector('.sidebar').classList.remove('mobile-open');
                        mobileOverlay.classList.remove('active');
                    }
                });
            });
            
            // Voltar ao Dashboard
            backToDashboardBtn.addEventListener('click', function() {
                mesariosSystem.classList.add('hidden');
                document.querySelector('main').classList.remove('hidden');
                document.body.classList.remove('overflow-hidden');
            });
            
            // Pesquisa e filtros
            searchMesariosInput.addEventListener('input', renderMesariosTable);
            filterMunicipioSelect.addEventListener('change', renderMesariosTable);
            
            // Paginação
            prevPageBtn.addEventListener('click', function() {
                if (currentPage > 1) {
                    currentPage--;
                    renderMesariosTable();
                }
            });
            
            nextPageBtn.addEventListener('click', function() {
                const totalPages = Math.ceil(getFilteredMesarios().length / itemsPerPage);
                if (currentPage < totalPages) {
                    currentPage++;
                    renderMesariosTable();
                }
            });
            
            // Importação de dados
            btnImport.addEventListener('click', importMesariosData);
            
            // Modal de edição
            cancelEditBtn.addEventListener('click', function() {
                editMesarioModal.classList.remove('active');
            });
            
            editMesarioForm.addEventListener('submit', function(e) {
                e.preventDefault();
                saveEditedMesario();
            });
        }
        
        function getFilteredMesarios() {
            let filtered = mesarios;
            
            // Filtro por pesquisa
            const searchTerm = searchMesariosInput.value.toLowerCase();
            if (searchTerm) {
                filtered = filtered.filter(mesario => 
                    mesario.nome.toLowerCase().includes(searchTerm) ||
                    mesario.funcao.toLowerCase().includes(searchTerm) ||
                    mesario.local_trabalho.toLowerCase().includes(searchTerm)
                );
            }
            
            // Filtro por município
            const municipioFilter = filterMunicipioSelect.value;
            if (municipioFilter) {
                filtered = filtered.filter(mesario => mesario.municipio === municipioFilter);
            }
            
            return filtered;
        }
        
        function renderMesariosTable() {
            const filteredMesarios = getFilteredMesarios();
            const startIndex = (currentPage - 1) * itemsPerPage;
            const endIndex = startIndex + itemsPerPage;
            const paginatedMesarios = filteredMesarios.slice(startIndex, endIndex);
            
            mesariosTableBody.innerHTML = '';
            
            if (paginatedMesarios.length === 0) {
                mesariosTableBody.innerHTML = `
                    <tr>
                        <td colspan="6" class="text-center py-4 text-gray-500">
                            Nenhum mesário encontrado
                        </td>
                    </tr>
                `;
            } else {
                paginatedMesarios.forEach(mesario => {
                    const row = document.createElement('tr');
                    row.innerHTML = `
                        <td>${mesario.nome}</td>
                        <td>${mesario.inscricao}</td>
                        <td>${mesario.funcao}</td>
                        <td>${mesario.local_trabalho}</td>
                        <td>${mesario.municipio}</td>
                        <td>
                            <button class="btn-marcar-faltoso text-yellow-600 hover:text-yellow-800 mr-2" data-id="${mesario.id}">
                                <i class="fas fa-user-times"></i>
                            </button>
                            <button class="btn-pedido-folga text-blue-600 hover:text-blue-800 mr-2" data-id="${mesario.id}">
                                <i class="fas fa-calendar-minus"></i>
                            </button>
                            <button class="btn-editar text-green-600 hover:text-green-800 mr-2" data-id="${mesario.id}">
                                <i class="fas fa-edit"></i>
                            </button>
                            <button class="btn-excluir text-red-600 hover:text-red-800" data-id="${mesario.id}">
                                <i class="fas fa-trash"></i>
                            </button>
                        </td>
                    `;
                    mesariosTableBody.appendChild(row);
                });
                
                // Adicionar event listeners aos botões
                document.querySelectorAll('.btn-editar').forEach(btn => {
                    btn.addEventListener('click', function() {
                        const id = this.getAttribute('data-id');
                        openEditModal(id);
                    });
                });
                
                document.querySelectorAll('.btn-excluir').forEach(btn => {
                    btn.addEventListener('click', function() {
                        const id = this.getAttribute('data-id');
                        openDeleteMesarioModal(id);
                    });
                });
            }
            
            // Atualizar controles de paginação
            const totalPages = Math.ceil(filteredMesarios.length / itemsPerPage);
            pageInfoSpan.textContent = `Página ${currentPage} de ${totalPages}`;
            
            prevPageBtn.disabled = currentPage === 1;
            nextPageBtn.disabled = currentPage === totalPages || totalPages === 0;
        }
        
        function openEditModal(id) {
            const mesario = mesarios.find(m => m.id == id);
            if (mesario) {
                currentEditingId = id;
                editMesarioForm.querySelector('input[name="nome"]').value = mesario.nome;
                editMesarioForm.querySelector('input[name="inscricao"]').value = mesario.inscricao;
                editMesarioForm.querySelector('input[name="funcao"]').value = mesario.funcao;
                editMesarioForm.querySelector('input[name="local_trabalho"]').value = mesario.local_trabalho;
                editMesarioModal.classList.add('active');
            }
        }
        
        function saveEditedMesario() {
            if (currentEditingId) {
                const index = mesarios.findIndex(m => m.id == currentEditingId);
                if (index !== -1) {
                    mesarios[index].nome = editMesarioForm.querySelector('input[name="nome"]').value;
                    mesarios[index].inscricao = editMesarioForm.querySelector('input[name="inscricao"]').value;
                    mesarios[index].funcao = editMesarioForm.querySelector('input[name="funcao"]').value;
                    mesarios[index].local_trabalho = editMesarioForm.querySelector('input[name="local_trabalho"]').value;
                    
                    localStorage.setItem('mesariosData', JSON.stringify(mesarios));
                    renderMesariosTable();
                    editMesarioModal.classList.remove('active');
                    currentEditingId = null;
                }
            }
        }
        
        function openDeleteMesarioModal(id) {
            const mesario = mesarios.find(m => m.id == id);
            if (mesario) {
                if (confirm(`Tem certeza que deseja excluir o mesário "${mesario.nome}"?`)) {
                    mesarios = mesarios.filter(m => m.id != id);
                    localStorage.setItem('mesariosData', JSON.stringify(mesarios));
                    renderMesariosTable();
                }
            }
        }
        
        function importMesariosData() {
            const municipio = municipioImportSelect.value;
            const file = fileImportInput.files[0];
            
            if (!municipio) {
                alert('Por favor, selecione um município.');
                return;
            }
            
            if (!file) {
                alert('Por favor, selecione um arquivo para importar.');
                return;
            }
            
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result);
                    const workbook = XLSX.read(data, { type: 'array' });
                    const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
                    const jsonData = XLSX.utils.sheet_to_json(firstSheet);
                    
                    // Processar dados (exemplo simplificado)
                    const novosMesarios = jsonData.map((row, index) => {
                        return {
                            id: mesarios.length > 0 ? Math.max(...mesarios.map(m => m.id)) + index + 1 : index + 1,
                            nome: row.NOME || row.nome || 'Nome não informado',
                            inscricao: row.INSCRICAO || row.inscricao || 'Não informada',
                            funcao: row.FUNCAO || row.funcao || 'Não informada',
                            local_trabalho: row['LOCAL DO TRABALHO'] || row.local_trabalho || 'Não informado',
                            municipio: municipio
                        };
                    });
                    
                    // Adicionar aos dados existentes
                    mesarios = [...mesarios, ...novosMesarios];
                    localStorage.setItem('mesariosData', JSON.stringify(mesarios));
                    
                    // Atualizar a tabela
                    renderMesariosTable();
                    
                    // Mostrar resultado
                    importResultDiv.innerHTML = `
                        <div class="bg-green-100 border border-green-400 text-green-700 px-4 py-3 rounded">
                            <strong>Sucesso!</strong> ${novosMesarios.length} mesários importados para ${municipio}.
                        </div>
                    `;
                    importResultDiv.classList.remove('hidden');
                    
                    // Limpar formulário
                    fileImportInput.value = '';
                    
                } catch (error) {
                    console.error('Erro na importação:', error);
                    importResultDiv.innerHTML = `
                        <div class="bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded">
                            <strong>Erro!</strong> Não foi possível importar o arquivo. Verifique o formato.
                        </div>
                    `;
                    importResultDiv.classList.remove('hidden');
                }
            };
            
            reader.readAsArrayBuffer(file);
        }
